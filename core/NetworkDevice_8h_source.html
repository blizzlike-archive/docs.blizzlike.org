204  template<typename T> static void serializeMessage
205  (uint32 t, const T& m, BinaryOutput& b) {
206 
207  b.writeUInt32(t);
208 
209  // Reserve space for the 4 byte size header
210  b.writeUInt32(0);
211 
212  size_t L = b.length();
213  m.serialize(b);
214  if ((size_t)b.length() == L) {
215  // No data was created by serialization.
216  // We need to send at least one byte because receive assumes that
217  // a zero length message is an error.
218  b.writeUInt8(0xFF);
219  }
220 
221  uint32 len = b.size() - 8;
222 
223  // We send the length first to tell recv how much data to read.
224  // Here we abuse BinaryOutput a bit and write directly into
225  // its buffer, violating the abstraction.

227  // the size field.
