42 // Legend:
43 // source - the type of object which executes the command
44 // target - the type of object used as target of the command if needed
45 // provided source - the "WorldObject* source" provided to the command
46 // provided target - the "WorldObject* target" provided to the command
47 
48 // For example the talk command can be used by any WorldObject and the text will
49 // be said by this object. If there is also a target provided of type Unit, then
50 // any use of $n in the text will show the name of this unit.
51 
52 // Command functions are defined in Map.h and implemented in ScriptCommands.cpp!

865 // Returns a target based on the type specified.
866 inline WorldObject* GetTargetByType(WorldObject* pSource, WorldObject* pTarget, uint8 TargetType, uint32 Param1 = 0u, uint32 Param2 = 0u)
867 {
868  switch (TargetType)
869  {
870  case TARGET_T_PROVIDED_TARGET:
871  return pTarget;
872  case TARGET_T_HOSTILE:
873  if (Unit* pUnitSource = ToUnit(pSource))
874  return pUnitSource->getVictim();
875  break;
876  case TARGET_T_HOSTILE_SECOND_AGGRO:
877  if (Creature* pCreatureSource = ToCreature(pSource))
878  return pCreatureSource->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 1);
879  break;
880  case TARGET_T_HOSTILE_LAST_AGGRO:
881  if (Creature* pCreatureSource = ToCreature(pSource))
882  return pCreatureSource->SelectAttackingTarget(ATTACKING_TARGET_BOTTOMAGGRO, 0);
883  break;
884  case TARGET_T_HOSTILE_RANDOM:
885  if (Creature* pCreatureSource = ToCreature(pSource))
886  return pCreatureSource->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0);
887  break;
888  case TARGET_T_HOSTILE_RANDOM_NOT_TOP:
889  if (Creature* pCreatureSource = ToCreature(pSource))
890  return pCreatureSource->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1);
891  break;
892  case TARGET_T_OWNER_OR_SELF:
893  if (Unit* pUnitSource = ToUnit(pSource))
894  return pUnitSource->GetCharmerOrOwnerOrSelf();
895  break;
896  case TARGET_T_OWNER:
897  if (Unit* pUnitSource = ToUnit(pSource))
898  return pUnitSource->GetOwner();
899  break;
900  case TARGET_T_FRIENDLY:
901  if (Unit* pUnitSource = ToUnit(pSource))
902  return pUnitSource->SelectRandomFriendlyTarget(Param2 ? ToUnit(pTarget) : nullptr, Param1 ? Param1 : 30.0f, true);
903  break;
904  case TARGET_T_FRIENDLY_INJURED:
905  if (Creature* pCreatureSource = ToCreature(pSource))
906  return pCreatureSource->DoSelectLowestHpFriendly(Param1 ? Param1 : 30.0f, Param2 ? Param2 : 50, true);
907  break;
908  case TARGET_T_FRIENDLY_INJURED_EXCEPT:
909  if (Creature* pCreatureSource = ToCreature(pSource))
910  return pCreatureSource->DoSelectLowestHpFriendly(Param1 ? Param1 : 30.0f, Param2 ? Param2 : 50, true, ToUnit(pTarget));
911  break;
912  case TARGET_T_FRIENDLY_MISSING_BUFF:
913  if (Creature* pCreatureSource = ToCreature(pSource))
914  return pCreatureSource->DoFindFriendlyMissingBuff(Param1 ? Param1 : 30.0f, Param2);
915  break;
916  case TARGET_T_FRIENDLY_MISSING_BUFF_EXCEPT:
917  if (Creature* pCreatureSource = ToCreature(pSource))
918  return pCreatureSource->DoFindFriendlyMissingBuff(Param1 ? Param1 : 30.0f, Param2, ToUnit(pTarget));
919  break;
920  case TARGET_T_FRIENDLY_CC:
921  if (Creature* pCreatureSource = ToCreature(pSource))
922  return pCreatureSource->DoFindFriendlyCC(Param1 ? Param1 : 30.0f);
923  break;
924  }
925  return nullptr;
926 }
