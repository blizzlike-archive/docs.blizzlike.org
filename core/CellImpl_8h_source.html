63 template<class T, class CONTAINER>
64 inline void
65 Cell::Visit(const CellPair &standing_cell, TypeContainerVisitor<T, CONTAINER> &visitor, Map &m, float x, float y, float radius) const
66 {
67  if (standing_cell.x_coord >= TOTAL_NUMBER_OF_CELLS_PER_MAP || standing_cell.y_coord >= TOTAL_NUMBER_OF_CELLS_PER_MAP)
68  return;
69 
70  //no jokes here... Actually placing ASSERT() here was good idea, but
71  //we had some problems with DynamicObjects, which pass radius = 0.0f (DB issue?)
72  //maybe it is better to just return when radius <= 0.0f?
73  if(radius <= 0.0f)
74  {
75  m.Visit(*this, visitor);
76  return;
77  }
78  //lets limit the upper value for search radius
79  if(radius > 333.0f)
80  radius = 333.0f;
81 
82  //lets calculate object coord offsets from cell borders.
83  CellArea area = Cell::CalculateCellArea(x, y, radius);
84  //if radius fits inside standing cell
85  if(!area)
86  {
87  m.Visit(*this, visitor);
88  return;
89  }
90 
91  CellPair &begin_cell = area.low_bound;
92  CellPair &end_cell = area.high_bound;
93  //visit all cells, found in CalculateCellArea()
94  //if radius is known to reach cell area more than 4x4 then we should call optimized VisitCircle
95  //currently this technique works with MAX_NUMBER_OF_CELLS 16 and higher, with lower values
96  //there are nothing to optimize because SIZE_OF_GRID_CELL is too big...
97  if(((end_cell.x_coord - begin_cell.x_coord) > 4) && ((end_cell.y_coord - begin_cell.y_coord) > 4))
98  {
99  VisitCircle(visitor, m, begin_cell, end_cell);
100  return;
101  }
102 
103  //ALWAYS visit standing cell first!!! Since we deal with small radiuses
104  //it is very essential to call visitor for standing cell firstly...
105  m.Visit(*this, visitor);
106 
107  // loop the cell range
108  for(uint32 x = begin_cell.x_coord; x <= end_cell.x_coord; x++)
109  {
110  for(uint32 y = begin_cell.y_coord; y <= end_cell.y_coord; y++)
111  {
112  CellPair cell_pair(x,y);
113  //lets skip standing cell since we already visited it
114  if(cell_pair != standing_cell)
115  {
116  Cell r_zone(cell_pair);
117  r_zone.data.Part.nocreate = data.Part.nocreate;
118  m.Visit(r_zone, visitor);
119  }
120  }
121  }
122 }

124 template<class T, class CONTAINER>
125 inline void
126 Cell::VisitCircle(TypeContainerVisitor<T, CONTAINER> &visitor, Map &m, const CellPair& begin_cell, const CellPair& end_cell) const
127 {
128  //here is an algorithm for 'filling' circum-squared octagon
129  uint32 x_shift = (uint32)ceilf((end_cell.x_coord - begin_cell.x_coord) * 0.3f - 0.5f);
130  //lets calculate x_start/x_end coords for central strip...
131  const uint32 x_start = begin_cell.x_coord + x_shift;
132  const uint32 x_end = end_cell.x_coord - x_shift;
133 
134  //visit central strip with constant width...
135  for(uint32 x = x_start; x <= x_end; ++x)
136  {
137  for(uint32 y = begin_cell.y_coord; y <= end_cell.y_coord; ++y)
138  {
139  CellPair cell_pair(x,y);
140  Cell r_zone(cell_pair);
141  r_zone.data.Part.nocreate = data.Part.nocreate;
142  m.Visit(r_zone, visitor);
143  }
144  }
145 
146  //if x_shift == 0 then we have too small cell area, which were already
147  //visited at previous step, so just return from procedure...
148  if(x_shift == 0)
149  return;
150 
151  uint32 y_start = end_cell.y_coord;
152  uint32 y_end = begin_cell.y_coord;
153  //now we are visiting borders of an octagon...
154  for (uint32 step = 1; step <= (x_start - begin_cell.x_coord); ++step)
155  {
156  //each step reduces strip height by 2 cells...
157  y_end += 1;
158  y_start -= 1;
159  for (uint32 y = y_start; y >= y_end; --y)
160  {
161  //we visit cells symmetrically from both sides, heading from center to sides and from up to bottom
162  //e.g. filling 2 trapezoids after filling central cell strip...
163  CellPair cell_pair_left(x_start - step, y);
164  Cell r_zone_left(cell_pair_left);
165  r_zone_left.data.Part.nocreate = data.Part.nocreate;
166  m.Visit(r_zone_left, visitor);
167 
168  //right trapezoid cell visit
169  CellPair cell_pair_right(x_end + step, y);
170  Cell r_zone_right(cell_pair_right);
171  r_zone_right.data.Part.nocreate = data.Part.nocreate;
172  m.Visit(r_zone_right, visitor);
173  }
174  }
175 }
