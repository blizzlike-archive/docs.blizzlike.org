231  inline Quat log() const {
232  if ((x == 0) && (y == 0) && (z == 0)) {
233  if (w > 0) {
234  return Quat(0, 0, 0, ::logf(w));
235  } else if (w < 0) {
236  // Log of a negative number. Multivalued, any number of the form
237  // (PI * v, ln(-q.w))
238  return Quat((float)pi(), 0, 0, ::logf(-w));
239  } else {
240  // log of zero!
241  return Quat((float)nan(), (float)nan(), (float)nan(), (float)nan());
242  }
243  } else {
244  // Partly imaginary.
245  float imagLen = sqrtf(x * x + y * y + z * z);
246  float len = sqrtf(imagLen * imagLen + w * w);
247  float theta = atan2f(imagLen, (float)w);
248  float t = theta / imagLen;
249  return Quat(t * x, t * y, t * z, ::logf(len));
250  }
251  }
