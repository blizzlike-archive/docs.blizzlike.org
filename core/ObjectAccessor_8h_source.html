50 template <class T>
51 class HashMapHolder
52 {
53  public:
54 
55  typedef UNORDERED_MAP<ObjectGuid, T*> MapType;
56  typedef ACE_RW_Thread_Mutex LockType;
57  typedef ACE_Read_Guard<LockType> ReadGuard;
58  typedef ACE_Write_Guard<LockType> WriteGuard;
59 
60  static void Insert(T* o)
61  {
62  WriteGuard guard(i_lock);
63  m_objectMap[o->GetObjectGuid()] = o;
64  }
65 
66  static void Remove(T* o)
67  {
68  WriteGuard guard(i_lock);
69  m_objectMap.erase(o->GetObjectGuid());
70  }
71 
72  static T* Find(ObjectGuid guid)
73  {
74  ReadGuard guard(i_lock);
75  typename MapType::iterator itr = m_objectMap.find(guid);
76  return (itr != m_objectMap.end()) ? itr->second : NULL;
77  }
78 
79  static MapType& GetContainer() { return m_objectMap; }
80 
81  static LockType& GetLock() { return i_lock; }
82
