314 inline wchar_t wcharToUpper(wchar_t wchar)
315 {
316  if(wchar >= L'a' && wchar <= L'z') // LATIN SMALL LETTER A - LATIN SMALL LETTER Z
317  return wchar_t(uint16(wchar)-0x0020);
318  if(wchar == 0x00DF) // LATIN SMALL LETTER SHARP S
319  return wchar_t(0x1E9E);
320  if(wchar >= 0x00E0 && wchar <= 0x00F6) // LATIN SMALL LETTER A WITH GRAVE - LATIN SMALL LETTER O WITH DIAERESIS
321  return wchar_t(uint16(wchar)-0x0020);
322  if(wchar >= 0x00F8 && wchar <= 0x00FE) // LATIN SMALL LETTER O WITH STROKE - LATIN SMALL LETTER THORN
323  return wchar_t(uint16(wchar)-0x0020);
324  if(wchar >= 0x0101 && wchar <= 0x012F) // LATIN SMALL LETTER A WITH MACRON - LATIN SMALL LETTER I WITH OGONEK (only %2=1)
325  {
326  if(wchar % 2 == 1)
327  return wchar_t(uint16(wchar)-0x0001);
328  }
329  if(wchar >= 0x0430 && wchar <= 0x044F) // CYRILLIC SMALL LETTER A - CYRILLIC SMALL LETTER YA
330  return wchar_t(uint16(wchar)-0x0020);
331  if(wchar == 0x0451) // CYRILLIC SMALL LETTER IO
332  return wchar_t(0x0401);
333 
334  return wchar;
335 }
