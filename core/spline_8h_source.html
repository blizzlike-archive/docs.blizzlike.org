182  template<class T> inline void initLengths(T& cacher)
183  {
184  index_type i = index_lo;
185  lengths.resize(index_hi + 1);
186  length_type prev_length = 0, new_length = 0;
187  while (i < index_hi)
188  {
189  new_length = cacher(*this, i);
190  // Length overflowed : assign max value
191  if (new_length < 0)
192  new_length = std::numeric_limits<length_type>::max();
193  lengths[++i] = new_length;
194 
195  MANGOS_ASSERT(prev_length <= new_length);
196  prev_length = new_length;
197  }
198  }
199

203  length_type length(index_type first, index_type last) const { return lengths[last] - lengths[first];}
