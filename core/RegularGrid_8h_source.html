139  template<typename RayCallback>
140  void intersectRay(const Ray& ray, RayCallback& intersectCallback, float& max_dist, const Vector3& end)
141  {
142  Cell cell = Cell::ComputeCell(ray.origin().x, ray.origin().y);
143  if (!cell.isValid())
144  return;
145 
146  Cell last_cell = Cell::ComputeCell(end.x, end.y);
147 
148  if (cell == last_cell)
149  {
150  if (Node* node = nodes[cell.x][cell.y])
151  node->intersectRay(ray, intersectCallback, max_dist);
152  return;
153  }
154 
155  float voxel = (float)CELL_SIZE;
156  float kx_inv = ray.invDirection().x, bx = ray.origin().x;
157  float ky_inv = ray.invDirection().y, by = ray.origin().y;
158 
159  int stepX, stepY;
160  float tMaxX, tMaxY;
161  if (kx_inv >= 0)
162  {
163  stepX = 1;
164  float x_border = (cell.x + 1) * voxel;
165  tMaxX = (x_border - bx) * kx_inv;
166  }
167  else
168  {
169  stepX = -1;
170  float x_border = (cell.x - 1) * voxel;
171  tMaxX = (x_border - bx) * kx_inv;
172  }
173 
174  if (ky_inv >= 0)
175  {
176  stepY = 1;
177  float y_border = (cell.y + 1) * voxel;
178  tMaxY = (y_border - by) * ky_inv;
179  }
180  else
181  {
182  stepY = -1;
183  float y_border = (cell.y - 1) * voxel;
184  tMaxY = (y_border - by) * ky_inv;
185  }
186 
187  //int Cycles = std::max((int)ceilf(max_dist/tMaxX),(int)ceilf(max_dist/tMaxY));
188  //int i = 0;
189 
190  float tDeltaX = voxel * fabs(kx_inv);
191  float tDeltaY = voxel * fabs(ky_inv);
192  do
193  {
194  if (Node* node = nodes[cell.x][cell.y])
195  {
196  //float enterdist = max_dist;
197  node->intersectRay(ray, intersectCallback, max_dist);
198  }
199  if (cell == last_cell)
200  break;
201  if (tMaxX < tMaxY)
202  {
203  tMaxX += tDeltaX;
204  cell.x += stepX;
205  }
206  else
207  {
208  tMaxY += tDeltaY;
209  cell.y += stepY;
210  }
211  //++i;
212  }
213  while (cell.isValid());
214  }
