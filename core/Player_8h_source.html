182  {
183  uint32 newData = action | (uint32(type) << 24);
184  if (newData != packedData || uState == ACTIONBUTTON_DELETED)
185  {
186  packedData = newData;
187  if (uState != ACTIONBUTTON_NEW)
188  uState = ACTIONBUTTON_CHANGED;
189  }
190  }
