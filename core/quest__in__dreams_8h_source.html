151 class InDreamsInterface
152 {
153 public:
154  InDreamsInterface()
155  {
156  m_uiGuids = new ObjectGuid[3];
157  }
158 
159  ~InDreamsInterface()
160  {
161  if (m_uiGuids)
162  delete m_uiGuids;
163  }
164 
165  ObjectGuid* m_uiGuids;
166 
167  Creature* GetCreature(Unit* pSource, InDreamsIndex index) const
168  {
169  return pSource ? pSource->GetMap()->GetCreature(m_uiGuids[index]) : nullptr;
170  }
171 
172  void AddGuid(ObjectGuid objectGuid, InDreamsIndex index) const
173  {
174  m_uiGuids[index] = objectGuid;
175  }
176 
177  template <typename AI>
178  AI* GetAI(Unit* pSource, InDreamsIndex index) const
179  {
180  if (auto pCreature = GetCreature(pSource, index))
181  {
182  if (auto pCreatureAI = static_cast<AI*>(pCreature->AI()))
183  return pCreatureAI;
184  }
185 
186  return nullptr;
187  }
