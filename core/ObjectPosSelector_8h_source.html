61  bool CheckAngle(UsedPosList::value_type const& nextUsedPos, float sign, float angle ) const
62  {
63  float angle_step2 = GetAngle(nextUsedPos.second);
64 
65  float next_angle = nextUsedPos.first;
66  if(nextUsedPos.second.sign * sign < 0) // last node from diff. list (-pi+alpha)
67  next_angle = 2.0f*M_PI_F-next_angle; // move to positive
68 
69  return fabs(angle)+angle_step2 <= next_angle;
70  }

80  bool NextAngleFor( UsedPosList::value_type const& usedPos, float sign, UsedPosType uptype, float &angle )
81  {
82  float angle_step = GetAngle(usedPos.second);
83 
84  // next possible angle
85  angle = usedPos.first * usedPos.second.sign + angle_step * sign;
86 
87  UsedPosList::value_type const* nextNode = nextUsedPos(uptype);
88  if(nextNode)
89  {
90  // if next node permit use selected angle, then do it
91  if(!CheckAngle(*nextNode, sign, angle))
92  {
93  m_smallStepOk[uptype] = false;
94  return false;
95  }
96  }
97 
98  // possible more points
99  m_smallStepOk[uptype] = true;
100  m_smallStepAngle[uptype] = angle;
101  m_smallStepNextUsedPos[uptype] = nextNode;
102 
103  return true;
104  }

106  bool NextSmallStepAngle( float sign, UsedPosType uptype, float &angle )
107  {
108  // next possible angle
109  angle = m_smallStepAngle[uptype] + m_anglestep * sign;
110 
111  if(fabs(angle) > M_PI)
112  {
113  m_smallStepOk[uptype] = false;
114  return false;
115  }
116 
117  if(m_smallStepNextUsedPos[uptype])
118  {
119  if(fabs(angle) >= m_smallStepNextUsedPos[uptype]->first)
120  {
121  m_smallStepOk[uptype] = false;
122  return false;
123  }
124 
125  // if next node permit use selected angle, then do it
126  if(!CheckAngle(*m_smallStepNextUsedPos[uptype], sign, angle))
127  {
128  m_smallStepOk[uptype] = false;
129  return false;
130  }
131  }
132 
133  // possible more points
134  m_smallStepAngle[uptype] = angle;
135  return true;
136  }
