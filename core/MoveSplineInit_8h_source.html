123  inline void MoveSplineInit::MoveTo(const Vector3& dest, uint32 options)
124  {
125  if (options & MOVE_PATHFINDING)
126  {
127  PathFinder path(&unit);
128  if (options & MOVE_EXCLUDE_STEEP_SLOPES)
129  path.ExcludeSteepSlopes();
130  path.setUseStrightPath(options & MOVE_STRAIGHT_PATH);
131  path.calculate(dest.x, dest.y, dest.z, options & MOVE_FORCE_DESTINATION);
132  Move(&path);
133  }
134  else
135  {
136  args.path_Idx_offset = 0;
137  args.path.resize(2);
138  args.path[1] = dest;
139  }
140  }
