31 <
32 typename T,
33 class ThreadingModel,
34 class CreatePolicy,
35 class LifeTimePolicy
36 >
37 T& MaNGOS::Singleton<T, ThreadingModel, CreatePolicy, LifeTimePolicy>::Instance()
38 {
39  if (!si_instance)
40  {
41  // double-checked Locking pattern
42  Guard();
43 
44  if (!si_instance)
45  {
46  if (si_destroyed)
47  {
48  si_destroyed = false;
49  LifeTimePolicy::OnDeadReference();
50  }
51 
52  si_instance = CreatePolicy::Create();
53  LifeTimePolicy::ScheduleCall(&DestroySingleton);
54  }
55  }
56 
57  return *si_instance;
58 }
