238 inline float Ray::intersectionTime(
239  const Vector3& vert0,
240  const Vector3& vert1,
241  const Vector3& vert2,
242  const Vector3& edge1,
243  const Vector3& edge2) const {
244 
245  (void)vert1;
246  (void)vert2;
247 
248  // Barycenteric coords
249  float u, v;
250 
251  float tvec[3], pvec[3], qvec[3];
252 
253  // begin calculating determinant - also used to calculate U parameter
254  CROSS(pvec, m_direction, edge2);
255 
256  // if determinant is near zero, ray lies in plane of triangle
257  const float det = DOT(edge1, pvec);
258 
259  if (det < EPSILON) {
260  return finf();
261  }
262 
263  // calculate distance from vert0 to ray origin
264  SUB(tvec, m_origin, vert0);
265 
266  // calculate U parameter and test bounds
267  u = DOT(tvec, pvec);
268  if ((u < 0.0f) || (u > det)) {
269  // Hit the plane outside the triangle
270  return finf();
271  }
272 
273  // prepare to test V parameter
274  CROSS(qvec, tvec, edge1);
275 
276  // calculate V parameter and test bounds
277  v = DOT(m_direction, qvec);
278  if ((v < 0.0f) || (u + v > det)) {
279  // Hit the plane outside the triangle
280  return finf();
281  }
282 
283 
284  // Case where we don't need correct (u, v):
285  const float t = DOT(edge2, qvec);
286 
287  if (t >= 0.0f) {
288  // Note that det must be positive
289  return t / det;
290  } else {
291  // We had to travel backwards in time to intersect
292  return finf();
293  }
294 }
