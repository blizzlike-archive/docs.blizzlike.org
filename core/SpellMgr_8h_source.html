147 {

154  case SPELL_EFFECT_SCHOOL_DAMAGE:

159  case SPELL_EFFECT_HEAL:
160  direct = true;
161  break;
162  case SPELL_EFFECT_APPLY_AURA:
163  switch (spellInfo->EffectApplyAuraName[i])
164  {
165  case SPELL_AURA_PERIODIC_DAMAGE:
166  case SPELL_AURA_PERIODIC_HEAL:
167  case SPELL_AURA_PERIODIC_ENERGIZE:
168  case SPELL_AURA_OBS_MOD_HEALTH:
