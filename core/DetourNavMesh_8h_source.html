514  inline dtPolyRef encodePolyId(unsigned int salt, unsigned int it, unsigned int ip) const
515  {
516 #ifdef DT_POLYREF64
517  return ((dtPolyRef)salt << (DT_POLY_BITS+DT_TILE_BITS)) | ((dtPolyRef)it << DT_POLY_BITS) | (dtPolyRef)ip;
518 #else
519  return ((dtPolyRef)salt << (m_polyBits+m_tileBits)) | ((dtPolyRef)it << m_polyBits) | (dtPolyRef)ip;
520 #endif
521  }
522 
530  inline void decodePolyId(dtPolyRef ref, unsigned int& salt, unsigned int& it, unsigned int& ip) const
531  {
532 #ifdef DT_POLYREF64
533  const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
534  const dtPolyRef tileMask = ((dtPolyRef)1<<DT_TILE_BITS)-1;
535  const dtPolyRef polyMask = ((dtPolyRef)1<<DT_POLY_BITS)-1;
536  salt = (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
537  it = (unsigned int)((ref >> DT_POLY_BITS) & tileMask);
538  ip = (unsigned int)(ref & polyMask);
539 #else
540  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
541  const dtPolyRef tileMask = ((dtPolyRef)1<<m_tileBits)-1;
542  const dtPolyRef polyMask = ((dtPolyRef)1<<m_polyBits)-1;
543  salt = (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
544  it = (unsigned int)((ref >> m_polyBits) & tileMask);
545  ip = (unsigned int)(ref & polyMask);
546 #endif
547  }
548 
553  inline unsigned int decodePolyIdSalt(dtPolyRef ref) const
554  {
555 #ifdef DT_POLYREF64
556  const dtPolyRef saltMask = ((dtPolyRef)1<<DT_SALT_BITS)-1;
557  return (unsigned int)((ref >> (DT_POLY_BITS+DT_TILE_BITS)) & saltMask);
558 #else
559  const dtPolyRef saltMask = ((dtPolyRef)1<<m_saltBits)-1;
560  return (unsigned int)((ref >> (m_polyBits+m_tileBits)) & saltMask);
561 #endif
562  }

568  inline unsigned int decodePolyIdTile(dtPolyRef ref) const

573 #else
