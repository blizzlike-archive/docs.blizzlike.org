289  Control evaluate(float s) const {
290  debugAssertM(control.size() == time.size(), "Corrupt spline: wrong number of control points.");
291 
292  /*
293  @cite http://www.gamedev.net/reference/articles/article1497.asp
294  Derivation of basis matrix follows.
295 
296  Given control points with positions p[i] at times t[i], 0 <= i <= 3, find the position
297  at time t[1] <= s <= t[2].
298 
299  Let u = s - t[0]
300  Let U = [u^0 u^1 u^2 u^3] = [1 u u^2 u^3]
301  Let dt0 = t[0] - t[-1]
302  Let dt1 = t[1] - t[0]
303  Let dt2 = t[2] - t[1]
304  */
305 
306  // Index of the first control point (i.e., the u = 0 point)
307  int i = 0;
308  // Fractional part of the time
309  float u = 0;
310 
311  computeIndex(s, i, u);
312 
313  Control p[4];
314  float t[4];
315  getControls(i - 1, t, p, 4);
316  float dt0 = t[1] - t[0];
317  float dt1 = t[2] - t[1];
318  float dt2 = t[3] - t[2];
319 
320  static const Matrix4 basis = computeBasis();
321 
322  // Powers of u
323  Vector4 uvec((float)(u*u*u), (float)(u*u), (float)u, 1.0f);
324 
325  // Compute the weights on each of the control points.
326  const Vector4& weights = uvec * basis;
327 
328  // Compute the weighted sum of the neighboring control points.
329  Control sum;
330 
331  const Control& p0 = p[0];
332  const Control& p1 = p[1];
333  const Control& p2 = p[2];
334  const Control& p3 = p[3];
335 
336  const Control& dp0 = p1 + (p0*-1.0f);
337  const Control& dp1 = p2 + (p1*-1.0f);
338  const Control& dp2 = p3 + (p2*-1.0f);
339 
340  // The factor of 1/2 from averaging two time intervals is
341  // already factored into the basis
342 
343  // tan1 = (dp0 / dt0 + dp1 / dt1) * ((dt0 + dt1) * 0.5);
344  // The last term normalizes for unequal time intervals
345  float x = (dt0 + dt1) * 0.5f;
346  float n0 = x / dt0;
347  float n1 = x / dt1;
348  float n2 = x / dt2;
349  const Control& dp1n1 = dp1 * n1;
350  const Control& tan1 = dp0 * n0 + dp1n1;
351  const Control& tan2 = dp1n1 + dp2 * n2;
352 
353  sum =
354  tan1 * weights[0]+
355  p1 * weights[1] +
356  p2 * weights[2] +
357  tan2 * weights[3];
358 
359
