66 /* mpq archive header. */
67 typedef struct {
68  uint32_t mpq_magic; /* the 0x1A51504D ('MPQ\x1A') signature. */
69  uint32_t header_size; /* mpq archive header size. */
70  uint32_t archive_size; /* size of mpq archive. */
71  uint16_t version; /* 0000 for starcraft and broodwar. */
72  uint16_t block_size; /* size of file block is (512 * 2 ^ block size). */
73  uint32_t hash_table_offset; /* file position of mpq_hash. */
74  uint32_t block_table_offset; /* file position of mpq_block, each entry has 16 bytes. */
75  uint32_t hash_table_count; /* number of entries in hash table. */
76  uint32_t block_table_count; /* number of entries in the block table. */
77 } PACK_STRUCT mpq_header_s;
