211 
212  // Hang onto the old m_bucket array
213  Node** oldBucket = m_bucket;
214 
215  // Allocate a new m_bucket array with the new size
216  m_bucket = (Node**)alloc(sizeof(Node*) * newSize);

742  Entry& getCreateEntry(const Key& key, bool& created) {
743  created = false;
744 
745  if (m_numBuckets == 0) {
746  resize(10);
747  }
748 
749  size_t code = HashFunc::hashCode(key);
750  size_t b = code % m_numBuckets;
751 
752  // Go to the m_bucket
753  Node* n = m_bucket[b];
754 
755  // No m_bucket, so this must be the first
756  if (n == NULL) {
757  m_bucket[b] = Node::create(key, code, NULL, m_memoryManager);
758  ++m_size;
759  created = true;
760  return m_bucket[b]->entry;
761  }
762 
763  size_t bucketLength = 1;
764 
765  // Sometimes a bad hash code will cause all elements
766  // to collide. Detect this case and don't rehash when
767  // it occurs; nothing good will come from the rehashing.
768  bool allSameCode = true;
769 
770  // Try to find the node
771  do {
772  allSameCode = allSameCode && (code == n->hashCode);
773 
774  if ((code == n->hashCode) && EqualsFunc::equals(n->entry.key, key)) {
775  // This is the a pre-existing node
776  return n->entry;
777  }
778 
779  n = n->next;
780  ++bucketLength;
781  } while (n != NULL);
782 
783  const size_t maxBucketLength = 3;
784  // (Don't bother changing the size of the table if all entries
785  // have the same hashcode--they'll still collide)
786  if ((bucketLength > maxBucketLength) &&
787  ! allSameCode &&
788  (m_numBuckets < m_size * 15)) {
789 
790  // This m_bucket was really large; rehash if all elements
791  // don't have the same hashcode the number of buckets is
792  // reasonable.
793 
794  // Back off the scale factor as the number of buckets gets
795  // large
796  float f = 3.0f;
797  if (m_numBuckets > 1000000) {
798  f = 1.5f;
799  } else if (m_numBuckets > 100000) {
800  f = 2.0f;
801  }
802  int newSize = iMax((int)(m_numBuckets * f) + 1, (int)(m_size * f));
803  resize(newSize);
804  }
805 
806  // Not found; insert at the head.
807  b = code % m_numBuckets;
808  m_bucket[b] = Node::create(key, code, m_bucket[b], m_memoryManager);
809  ++m_size;
810  created = true;
811  return m_bucket[b]->entry;
812  }
