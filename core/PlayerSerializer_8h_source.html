300  if (isAlive() && item->IsLimitedToAnotherMapOrZone(GetMapId(), zone))
301  isValid = false;

327  item = BankItem(dest, item, true);
328  }
329 
330  // store bags that may contain items in them
331  if (item->IsBag() && IsBagPos(item->GetPos()))

333  }
334  // the item/bag in a bag
335  else
336  {

338  // the item is in a bag, find the bag
339  std::unordered_map<uint32, Bag*>::const_iterator itr = bagMap.find(bag_guid);
340  ASSERT(itr != bagMap.end());
341  ASSERT(item_slot < itr->second->GetBagSize());
342  ItemPosCountVec dest;
343  ASSERT(CanStoreItem(itr->second->GetSlot(), item_slot, dest, item, false) == EQUIP_ERR_OK);
344  item = StoreItem(dest, item, true);
345  }
346  }
347 
348  if (!isValid && item)
349  delete item;
350 
351  if (isValid && item)
352  {
353  // Do not update items for player
354  item->SetState(ITEM_UNCHANGED, this);
355  if (item->GetContainer())
356  item->GetContainer()->SetState(ITEM_UNCHANGED, this);
357  }
358  }
