150  static const AreaEntry* GetByAreaFlagAndMap(uint32 areaFlag, uint32 mapId)
151  {
152  AreaEntry const* areaEntry = nullptr;
153  // 1.12.1 areatable have duplicates for areaflag
154  for (auto itr = sAreaStorage.begin<AreaEntry>(); itr < sAreaStorage.end<AreaEntry>(); ++itr)
155  {
156  if (areaFlag && areaFlag == itr->ExploreFlag)
157  {
158  if (itr->MapId == mapId)
159  return *itr;
160 
161  areaEntry = *itr;
162  }
163  }
164 
165  if (areaEntry)
166  return areaEntry;
167 
168  if (const auto *mapEntry = sMapStorage.LookupEntry<MapEntry>(mapId))
169  return sAreaStorage.LookupEntry<AreaEntry>(mapEntry->linkedZone);
170 
171  return nullptr;
172  }
