196 inline SpellSchools GetFirstSchoolInMask(SpellSchoolMask mask)
197 {
198  for(int i = 0; i < MAX_SPELL_SCHOOL; ++i)
199  if(mask & (1 << i))
200  return SpellSchools(i);
201 
202  return SPELL_SCHOOL_NORMAL;
